private static void add(Node node, int val) {
		if (node == null) {
			node = new Node(val);
			System.out.println(val + " is added!");
		}
		int cmp = node.data.compareTo(val);
		if (cmp < 0) {
			add(node.rightChild,val);
		}
		else if (cmp > 0) {
			add(node.leftChild,val);
		}
	}
