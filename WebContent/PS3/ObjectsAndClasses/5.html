public class Building {
	//Following are the data fields
	public int doorNumber;
	public int nofApartments;
	public String nameOfSuper;
	
	public Apartment[] apartments;	//all apartments are kept

	//Following are the methods
	//First one is a special method, ignore it for the time being
	public Building(int inNofApartments){
		this.doorNumber = -1;
		this.nofApartments = inNofApartments;
		this.nameOfSuper = "";
		
		if (this.nofApartments > 0)	//validity check
			this.apartments = new Apartment[this.nofApartments];	//allocation
	}

	//Regular methods are below
	public void printDoorNumber(){
		System.out.println("Number of this building is: " + this.doorNumber);
	}
	
	public void fooBld(){
		System.out.println("FooBld called!");
	}

	public void barBld(){
		System.out.println("BarBld called!");
	}
}
