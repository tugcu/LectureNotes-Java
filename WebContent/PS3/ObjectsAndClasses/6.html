public class Neighborhood {
	//Following are the data fields
	public String name;
	public int nofBuildings;

	public Building[] buildings;	//all buildings are kept

	//Following are the methods
	//First one is a special method, ignore it for the time being
	public Neighborhood(int inNofBuildings){
		this.name = "";
		this.nofBuildings = inNofBuildings;

		if (this.nofBuildings > 0)	//validity check
			this.buildings = new Building[this.nofBuildings];	//allocation
	}

	//Regular methods are below
	public void printName(){
		System.out.println("Name of this neighborhood is: " + this.name);
	}

	public void fooBld(){
		System.out.println("FooNgb called!");
	}

	public void barBld(){
		System.out.println("BarNgb called!");
	}
}
