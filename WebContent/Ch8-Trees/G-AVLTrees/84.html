protected TreeNode add(TreeNode node, E element) {
	node = super.add(node, element);
	node = rebalance(node);
	return node;
}

protected TreeNode rebalance(TreeNode node) {
	int bf = balanceFactor(node);
	if (bf < -1) {
		if (balanceFactor(node.left) < 0) {   // case 1 (R)
			node = rightRotate(node);
		} else {                              // case 2 (LR)
			node.left = leftRotate(node.left);
			node = rightRotate(node);
		}
	} else if (bf > 1) {
		if (balanceFactor(node.right) < 0) {  // case 3 (RL)
			node.right = rightRotate(node.right);
			node = leftRotate(node);
		} else {                              // case 4 (L)
			node = leftRotate(node);
		}
	}
	return node;
}
