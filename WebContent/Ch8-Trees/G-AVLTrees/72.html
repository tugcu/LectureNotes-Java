Code for Right Rotation 

private TreeNode rightRotate(TreeNode parent) {
	// 1. detach left child's right subtree
	TreeNode leftright = parent.left.right;

	// 2. consider left child to be the new parent
	TreeNode newParent = parent.left;

	// 3. attach old parent onto right of new parent
	newParent.right = parent;

	// 4. attach old left child's old right subtree as left subtree of new right child
	newParent.right.left = leftright;

	return newParent;
}
